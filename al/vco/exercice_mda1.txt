exercice amelioration gigagallerie

1) facile/moyen
filtrer les images qui ne sont pas taggés par les tags choisis

select DISTINCT i 
FROM Image as i left join fetch i.tags, IN(i.tags) ta1, IN(i.tags) ta2 
WHERE ta1.id = :tp1 
	AND ta2.id = :tp2 
	AND NOT EXISTS ( 
		SELECT i2 
		FROM Image AS i2, IN(i2.tags) te1 
		WHERE ( te1.id = :te1 OR te1.id = :te2) 
			and i2.id = i.id 
		)
